            height: 0;
        }

            height: var(--radix-accordion-content-height);
        }
    }

    @keyframes accordion-up {
        from {
            height: var(--radix-accordion-content-height);
        }
        to {
            height: 0;
        }
    }
}`} />

If you set `type` to `multiple` then accordion will allow multiple items to expend the same time

## Props

### Accordion

| Prop    | Type                         | Default   |
|---------|------------------------------|-----------|
| variant | `underline` `outline` `soft` | underline |

### AccordionTrigger

| Prop | Type      | Default               |
|------|-----------|-----------------------|
| icon | ReactNode | `<ChevronDownIcon />` |
